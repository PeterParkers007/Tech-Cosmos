// {{HERO_TYPE}} {{HERO_NAMESPACE}}
Shader "Custom/SpriteOutline" 
{
    Properties
    {
        _MainTex ("Main Texture", 2D) = "white" {}
        _OutlineColor ("Outline Color", Color) = (1,0,0,1)
        _OutlineThickness ("Outline Thickness", Range(0.001, 10)) = 0.01
        _EdgeThreshold ("Edge Threshold", Range(0, 1)) = 0.5
    }
 
    SubShader
    {
        Tags { 
            "Queue"="Transparent" 
            "RenderType"="Transparent" 
            "IgnoreProjector"="True" 
        }
 
        Blend SrcAlpha OneMinusSrcAlpha
        Cull Off
        ZWrite Off
 
        Pass
        {
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            
            #include "UnityCG.cginc"
 
            struct appdata
            {
                float4 vertex : POSITION;
                float2 uv : TEXCOORD0;
            };
 
            struct v2f
            {
                float2 uv : TEXCOORD0;
                float4 vertex : SV_POSITION;
            };
 
            sampler2D _MainTex;
            float4 _MainTex_TexelSize; // 自动获取纹理像素尺寸
            float4 _MainTex_ST;
            float4 _OutlineColor;
            float _OutlineThickness;
            float _EdgeThreshold;
 
            v2f vert (appdata v)
            {
                v2f o;
                o.vertex = UnityObjectToClipPos(v.vertex);
                o.uv = TRANSFORM_TEX(v.uv, _MainTex);
                return o;
            }
 
            float is_edge(sampler2D tex, float2 uv, float2 pixel_size)
            {
                float center_alpha = tex2D(tex, uv).a;
                if (center_alpha > _EdgeThreshold) return 0; // 中心点不透明则不是边缘
                
                // 8方向采样检测
                float max_alpha = 0;
                for (int x = -1; x <= 1; x++) {
                    for (int y = -1; y <= 1; y++) {
                        if (x == 0 && y == 0) continue;
                        float2 offset = float2(x, y) * pixel_size * _OutlineThickness;
                        max_alpha = max(max_alpha, tex2D(tex, uv + offset).a);
                    }
                }
                return step(_EdgeThreshold, max_alpha); // 周围有实心像素则返回1
            }
 
            fixed4 frag (v2f i) : SV_Target
            {
                float2 pixel_size = _MainTex_TexelSize.xy; // 正确获取像素尺寸
                fixed4 col = tex2D(_MainTex, i.uv);
                
                float edge = is_edge(_MainTex, i.uv, pixel_size);
                fixed4 final_color = lerp(col, _OutlineColor, edge);
                final_color.a = max(col.a, edge * _OutlineColor.a); // 保持原始透明度或描边透明度
                
                return final_color;
            }
            ENDCG
        }
    }
}
